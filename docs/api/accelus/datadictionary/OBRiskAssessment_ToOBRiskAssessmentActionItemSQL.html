SQL Example for: Obligation Risk Assessment, Obligation Risk Assessment Action Item
SELECT OBRiskAssessment.DocumentID,OBRiskAssessmentActionItem.DocumentID
FROM
OBRiskAssessment
JOIN OBRiskAssessment_ToOBRiskAssessmentActionItem ON OBRiskAssessment.DocumentID = OBRiskAssessment_ToOBRiskAssessmentActionItem.DocumentID
JOIN OBRiskAssessmentActionItem ON OBRiskAssessmentActionItem.DocumentID = OBRiskAssessment_ToOBRiskAssessmentActionItem.ToOBRiskAssessmentActionItem
