SQL Example for: OSCP Risk Assessment, OSCP Risk Assessment Action Item
SELECT OSCPRiskAssessment.DocumentID,OSCPRiskAssessmentActionItem.DocumentID
FROM
OSCPRiskAssessment
JOIN OSCPRiskAssessment_ToOSCPRiskAssessmentActionItem ON OSCPRiskAssessment.DocumentID = OSCPRiskAssessment_ToOSCPRiskAssessmentActionItem.DocumentID
JOIN OSCPRiskAssessmentActionItem ON OSCPRiskAssessmentActionItem.DocumentID = OSCPRiskAssessment_ToOSCPRiskAssessmentActionItem.ToOSCPRiskAssessmentActionItem
